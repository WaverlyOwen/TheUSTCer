题目序列化记录（题目工坊 / #p= 分享链接，见 core/puzzle-io.js）：
{
  v: 1,
  w, h,                        棋盘尺寸
  sign,                        (w+1)×(h+1) 的 signMap
  answer: moves[] | null,      参考答案移动序列（可无）
  palette: [{color, chars[]}], 自定义色表
  roadNames: [string],         自定义路名表
  name, createdAt,
  origin: 'generated' | 'custom' | 'imported'
}
分享链接 = 页面地址 + '#p=' + base64url(JSON)
